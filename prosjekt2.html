LEGO-prosjekt 2

Formål

Bli kjent med trykksensor og ultralydsensor, samt kontrollere roboten med løkker og betingelser.

Kode


#!/usr/bin/env pybricks-micropython
from pybricks.hubs import EV3Brick
from pybricks.ev3devices import Motor, TouchSensor, UltrasonicSensor
from pybricks.parameters import Port
from pybricks.tools import wait


ev3 = EV3Brick()
venstre_motor = Motor(Port.C)
hoyre_motor = Motor(Port.B)
start_stopp_sensor = TouchSensor(Port.S1)
ultralyd_sensor = UltrasonicSensor(Port.S2)


driving = False
ev3.speaker.say("Ready")
